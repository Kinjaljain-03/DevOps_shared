#this is C program

n = int(input("Enter number of subjects: "))
total = 0

for i in range(n):
    marks = int(input(f"Enter marks of subject {i+1}: "))
    total += marks

average = total / n

print("Total Marks:", total)
print("Average Marks:", average)

if average >= 40:
    print("Result: PASS")
else:
    print("Result: FAIL")
